```c
int fun1() {
	fun2(int argv1);
	return 0;
}

int fun2(int argv1) {
	int a = 0, b = 1;
	int c = argv1 + fun3(a, b);
	return 0;
}

int fun3(int argv1, int argv2) {
	return argv1+ argv2;
}
```
